        if [[ "$line" =~ ^# ]]; then

        fi
        current_header="$line"

        if [ -n "$current_header" ]; then
            current_header="$current_header"$'\n'"$line"
        fi
    fi
done < "$input_file"

# Process the last header, if any
if [ -n "$current_header" ]; then
    extract_and_create_file "$current_header"
